int x = 7;
int y = 3;

if (x > y && x > 0) {
    print(1);
}

if (x == 7 || y == 5) {
    print(2);
}

if (!(x < y)) {
    print(3);
}

int i = 0;
while (i < 2) {
    print(10);
    i = i + 1;
}

    case 5:
        result = 100;

    case 7:
        result = 200;
        break;
    default:
        result = 0;

}
